terraform {
  required_version = ">= 1.5.0"

  required_providers {
    aws = {
      source  = "hashicorp/aws"
      version = "~> 5.0"
    }
  }

  # Production assumption: remote state via S3 + DynamoDB locking.
  # (Backend config is intentionally not hard-coded here.)
}

# Variables (no env hardcodes)
########################

variable "region" {
  type        = string
  description = "AWS region"
  default     = "eu-west-1"
}

variable "system_name" {
  type        = string
  description = "System/application name"
}

variable "environment" {
  type        = string
  description = "Environment name (e.g., dev, staging, prod)"
}

variable "vpc_cidr" {
  type        = string
  description = "VPC CIDR"
}

variable "az_count" {
  type        = number
  description = "How many AZs to span"
  default     = 2
}

variable "cw_logs_retention_days" {
  type        = number
  description = "CloudWatch Logs retention (must be >= 365 for Checkov CKV_AWS_338)"
  default     = 365

  validation {
    condition     = var.cw_logs_retention_days >= 365
    error_message = "cw_logs_retention_days must be >= 365 to satisfy CKV_AWS_338."
  }
}


variable "kms_admin_principal_arns" {
  type        = list(string)
  description = "List of IAM principal ARNs allowed to administer the CMK (recommended: a break-glass/admin role, not root)."
  default     = []
}

variable "instance_type" {
  type        = string
  description = "EC2 instance type"
  default     = "t3.micro"
}

variable "asg_desired" {
  type        = number
  description = "ASG desired capacity"
  default     = 1
}

variable "asg_min" {
  type        = number
  description = "ASG min capacity"
  default     = 1
}

variable "asg_max" {
  type        = number
  description = "ASG max capacity"
  default     = 2
}

########################
# Provider + data
########################

provider "aws" {
  region = var.region
}

data "aws_caller_identity" "current" {}

data "aws_region" "current" {}

data "aws_availability_zones" "available" {
  state = "available"
}

locals {
  name_prefix = "${var.system_name}-${var.environment}"
  azs         = slice(data.aws_availability_zones.available.names, 0, var.az_count)

  # Simple deterministic subnetting: two /20s per AZ (public + private)
  # Use non-overlapping CIDRs derived from the VPC CIDR.
  public_subnet_cidrs  = [for i in range(var.az_count) : cidrsubnet(var.vpc_cidr, 4, i)]
  private_subnet_cidrs = [for i in range(var.az_count) : cidrsubnet(var.vpc_cidr, 4, i + 8)]
}

# Least-privilege key policy:
# - No "kms:*".
# - Admin actions are explicitly enumerated.
# - Admin principals are provided via variable.
# - Services (Logs, Backup, EC2/EBS) get only required data-plane permissions.

data "aws_iam_policy_document" "kms_key" {
  statement {
    sid = "AllowKeyAdministration"

    principals {
      type        = "AWS"
      identifiers = length(var.kms_admin_principal_arns) > 0 ? var.kms_admin_principal_arns : ["arn:aws:iam::${data.aws_caller_identity.current.account_id}:root"]
    }

    actions = [
      "kms:Create*",
      "kms:Describe*",
      "kms:Enable*",
      "kms:List*",
      "kms:Put*",
      "kms:Update*",
      "kms:Revoke*",
      "kms:Disable*",
      "kms:Get*",
      "kms:Delete*",
      "kms:TagResource",
      "kms:UntagResource",
      "kms:ScheduleKeyDeletion",
      "kms:CancelKeyDeletion"
    ]

    # CKV_AWS_356: avoid Resource="*" for restrictable actions.
    resources = [aws_kms_key.primary.arn]
  }

  statement {
    sid = "AllowCloudWatchLogsUse"

    principals {
      type        = "Service"
      identifiers = ["logs.${data.aws_region.current.name}.amazonaws.com"]
    }

    actions = [
      "kms:Encrypt",
      "kms:Decrypt",
      "kms:ReEncrypt*",
      "kms:GenerateDataKey*",
      "kms:DescribeKey"
    ]

    resources = [aws_kms_key.primary.arn]

    condition {
      test     = "ArnLike"
      variable = "kms:EncryptionContext:aws:logs:arn"
      values   = ["arn:aws:logs:${data.aws_region.current.name}:${data.aws_caller_identity.current.account_id}:log-group:*"]
    }
  }

  statement {
    sid = "AllowBackupUse"

    principals {
      type        = "Service"
      identifiers = ["backup.amazonaws.com"]
    }

    actions = [
      "kms:Encrypt",
      "kms:Decrypt",
      "kms:ReEncrypt*",
      "kms:GenerateDataKey*",
      "kms:DescribeKey"
    ]

    resources = [aws_kms_key.primary.arn]
  }

  statement {
    sid = "AllowEC2EBSUse"

    principals {
      type        = "Service"
      identifiers = ["ec2.amazonaws.com"]
    }

    actions = [
      "kms:Encrypt",
      "kms:Decrypt",
      "kms:ReEncrypt*",
      "kms:GenerateDataKey*",
      "kms:DescribeKey",
      "kms:CreateGrant"
    ]

    resources = [aws_kms_key.primary.arn]

    condition {
      test     = "Bool"
      variable = "kms:GrantIsForAWSResource"
      values   = ["true"]
    }
  }
}


resource "aws_kms_key" "primary" {
  description             = "${local.name_prefix} primary CMK (EBS/Logs/Backup)"
  enable_key_rotation     = true
  deletion_window_in_days = 30
}

# Attach the key policy after the key exists (avoids circular reference and allows Resource to be key ARN).
resource "aws_kms_key_policy" "primary" {
  key_id = aws_kms_key.primary.key_id
  policy = data.aws_iam_policy_document.kms_key.json
}


resource "aws_kms_alias" "primary" {
  name          = "alias/${local.name_prefix}/primary"
  target_key_id = aws_kms_key.primary.key_id
}

resource "aws_ebs_encryption_by_default" "this" {
  enabled = true
}

resource "aws_ebs_default_kms_key" "this" {
  key_arn = aws_kms_key.primary.arn
}

########################
# Networking (VPC, subnets, routing)
########################

resource "aws_vpc" "this" {
  cidr_block           = var.vpc_cidr
  enable_dns_support   = true
  enable_dns_hostnames = true

  tags = {
    Name = "${local.name_prefix}-vpc"
  }
}

# CKV2_AWS_12 remediation: lock down default SG (no ingress/egress).
resource "aws_default_security_group" "this" {
  vpc_id = aws_vpc.this.id

  # No ingress/egress blocks means: Terraform will remove any existing rules.
  tags = {
    Name = "${local.name_prefix}-default-sg"
  }
}

resource "aws_internet_gateway" "this" {
  vpc_id = aws_vpc.this.id

  tags = {
    Name = "${local.name_prefix}-igw"
  }
}

resource "aws_subnet" "public" {
  for_each = { for idx, az in local.azs : idx => az }

  vpc_id                  = aws_vpc.this.id
  availability_zone       = each.value
  cidr_block              = local.public_subnet_cidrs[tonumber(each.key)]
  # CKV_AWS_130 remediation: do NOT auto-assign public IPs.
  map_public_ip_on_launch = false

  tags = {
    Name = "${local.name_prefix}-public-${each.value}"
    Tier = "public"
  }
}

resource "aws_subnet" "private" {
  for_each = { for idx, az in local.azs : idx => az }

  vpc_id                  = aws_vpc.this.id
  availability_zone       = each.value
  cidr_block              = local.private_subnet_cidrs[tonumber(each.key)]
  map_public_ip_on_launch = false

  tags = {
    Name = "${local.name_prefix}-private-${each.value}"
    Tier = "private"
  }
}

resource "aws_route_table" "public" {
  vpc_id = aws_vpc.this.id

  route {
    cidr_block = "0.0.0.0/0"
    gateway_id = aws_internet_gateway.this.id
  }

  tags = {
    Name = "${local.name_prefix}-public-rt"
  }
}

resource "aws_route_table_association" "public" {
  for_each = aws_subnet.public

  subnet_id      = each.value.id
  route_table_id = aws_route_table.public.id
}

# One NAT GW per AZ for resilience.
resource "aws_eip" "nat" {
  for_each = aws_subnet.public

  domain = "vpc"

  tags = {
    Name = "${local.name_prefix}-nat-eip-${each.key}"
  }
}

resource "aws_nat_gateway" "this" {
  for_each = aws_subnet.public

  allocation_id = aws_eip.nat[each.key].id
  subnet_id     = aws_subnet.public[each.key].id

  tags = {
    Name = "${local.name_prefix}-nat-${each.key}"
  }

  depends_on = [aws_internet_gateway.this]
}

resource "aws_route_table" "private" {
  for_each = aws_subnet.private

  vpc_id = aws_vpc.this.id

  tags = {
    Name = "${local.name_prefix}-private-rt-${each.key}"
  }
}

resource "aws_route" "private_nat" {
  for_each = aws_route_table.private

  route_table_id         = each.value.id
  destination_cidr_block = "0.0.0.0/0"
  nat_gateway_id         = aws_nat_gateway.this[each.key].id
}

resource "aws_route_table_association" "private" {
  for_each = aws_subnet.private

  subnet_id      = each.value.id
  route_table_id = aws_route_table.private[each.key].id
}

########################
# VPC Endpoints (SSM-only access; reduces public egress)
########################

# Instance SG: no inbound; egress only HTTPS (SSM, patching, endpoints)
resource "aws_security_group" "instance" {
  name        = "${local.name_prefix}-instance"
  description = "Instance SG: no inbound; egress minimal"
  vpc_id      = aws_vpc.this.id

  # No ingress rules: SSM Session Manager does not require inbound.

  egress {
    description = "HTTPS egress (prefer VPC endpoints; NAT for patching)"
    from_port   = 443
    to_port     = 443
    protocol    = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }

  tags = {
    Name = "${local.name_prefix}-instance-sg"
  }
}

# Endpoints SG: allow HTTPS from instances; restrict egress back to instances.
resource "aws_security_group" "endpoints" {
  name        = "${local.name_prefix}-vpce"
  description = "VPC endpoints security group"
  vpc_id      = aws_vpc.this.id

  ingress {
    description     = "HTTPS from instances"
    from_port       = 443
    to_port         = 443
    protocol        = "tcp"
    security_groups = [aws_security_group.instance.id]
  }

  # CKV_AWS_382 + CKV_AWS_23 remediation: no 0.0.0.0/0 all-protocol egress, add description.
  egress {
    description     = "Return traffic to instances over HTTPS"
    from_port       = 443
    to_port         = 443
    protocol        = "tcp"
    security_groups = [aws_security_group.instance.id]
  }

  tags = {
    Name = "${local.name_prefix}-vpce-sg"
  }
}

# Secure Compute: EC2 via ASG across private subnets (multi-AZ)

data "aws_ami" "al2023" {
  most_recent = true
  owners      = ["amazon"]

  filter {
    name   = "name"
    values = ["al2023-ami-*-x86_64"]
  }

  filter {
    name   = "virtualization-type"
    values = ["hvm"]
  }

  filter {
    name   = "architecture"
    values = ["x86_64"]
  }
}

data "aws_iam_policy_document" "ec2_assume" {
  statement {
    actions = ["sts:AssumeRole"]

    principals {
      type        = "Service"
      identifiers = ["ec2.amazonaws.com"]
    }
  }
}

resource "aws_iam_role" "ec2" {
  name               = "${local.name_prefix}-ec2"
  assume_role_policy = data.aws_iam_policy_document.ec2_assume.json
}

# SSM access without SSH (Session Manager).
resource "aws_iam_role_policy_attachment" "ssm_core" {
  role       = aws_iam_role.ec2.name
  policy_arn = "arn:aws:iam::aws:policy/AmazonSSMManagedInstanceCore"
}

resource "aws_iam_instance_profile" "ec2" {
  name = "${local.name_prefix}-ec2"
  role = aws_iam_role.ec2.name
}

resource "aws_launch_template" "this" {
  name_prefix   = "${local.name_prefix}-lt-"
  image_id      = data.aws_ami.al2023.id
  instance_type = var.instance_type

  iam_instance_profile {
    name = aws_iam_instance_profile.ec2.name
  }

  network_interfaces {
    # No public IPs (private-only)
    associate_public_ip_address = false
    security_groups             = [aws_security_group.instance.id]
  }

  metadata_options {
    http_tokens                 = "required" # IMDSv2
    http_endpoint               = "enabled"
    http_put_response_hop_limit = 1
  }

  block_device_mappings {
    device_name = "/dev/xvda"

    ebs {
      encrypted             = true
      kms_key_id            = aws_kms_key.primary.arn
      volume_type           = "gp3"
      delete_on_termination = true
    }
  }

  monitoring {
    enabled = true
  }

  tag_specifications {
    resource_type = "instance"

    tags = {
      Name = "${local.name_prefix}-ec2"
    }
  }

  tag_specifications {
    resource_type = "volume"

    tags = {
      Name = "${local.name_prefix}-ebs"
    }
  }
}

resource "aws_autoscaling_group" "this" {
  name                      = "${local.name_prefix}-asg"
  desired_capacity          = var.asg_desired
  min_size                  = var.asg_min
  max_size                  = var.asg_max
  health_check_type         = "EC2"
  health_check_grace_period = 300

  vpc_zone_identifier = [for s in aws_subnet.private : s.id]

  launch_template {
    id      = aws_launch_template.this.id
    version = "$Latest"
  }

  # Ensure instances are spread across AZs
  capacity_rebalance = true

  tag {
    key                 = "Name"
    value               = "${local.name_prefix}-asg"
    propagate_at_launch = true
  }
}
