resource "aws_kms_key" "main" {
  description             = "${local.name_prefix} primary CMK"
  deletion_window_in_days = 30
  enable_key_rotation     = true

  # Key policy must allow AWS services (EC2/EBS + CloudWatch Logs) to use the CMK.
  # Least-privilege is enforced via kms:ViaService + kms:CallerAccount + GrantIsForAWSResource.
  policy = jsonencode({
    Version = "2012-10-17"
    Statement = [
      # Admin (account root) â€” enables IAM policies in the account to be evaluated.
      {
        Sid    = "AllowAccountAdminsFullAccess"
        Effect = "Allow"
        Principal = {
          AWS = "arn:aws:iam::${data.aws_caller_identity.current.account_id}:root"
        }
        Action   = "kms:*"
        Resource = "*"
      },

      # EC2/EBS encryption for instances launched by ASG (root volume create/attach).
      {
        Sid    = "AllowEbsViaEc2InThisAccount"
        Effect = "Allow"
        Principal = {
          AWS = "*"
        }
        Action = [
          "kms:Encrypt",
          "kms:Decrypt",
          "kms:ReEncrypt*",
          "kms:GenerateDataKey*",
          "kms:DescribeKey"
        ]
        Resource = "*"
        Condition = {
          StringEquals = {
            "kms:CallerAccount" = data.aws_caller_identity.current.account_id
            "kms:ViaService"    = "ec2.${var.aws_region}.amazonaws.com"
          }
        }
      },
      {
        Sid    = "AllowEbsGrantsViaEc2InThisAccount"
        Effect = "Allow"
        Principal = {
          AWS = "*"
        }
        Action = [
          "kms:CreateGrant",
          "kms:ListGrants",
          "kms:RevokeGrant"
        ]
        Resource = "*"
        Condition = {
          Bool = {
            "kms:GrantIsForAWSResource" = "true"
          }
          StringEquals = {
            "kms:CallerAccount" = data.aws_caller_identity.current.account_id
            "kms:ViaService"    = "ec2.${var.aws_region}.amazonaws.com"
          }
        }
      },

      # CloudWatch Logs encryption for VPC Flow Logs log group.
      {
        Sid    = "AllowCloudWatchLogsUseOfKey"
        Effect = "Allow"
        Principal = {
          Service = "logs.${var.aws_region}.amazonaws.com"
        }
        Action = [
          "kms:Encrypt",
          "kms:Decrypt",
          "kms:ReEncrypt*",
          "kms:GenerateDataKey*",
          "kms:DescribeKey"
        ]
        Resource = "*"
      }
    ]
  })

  tags = {
    Name = "${local.name_prefix}-cmk"
  }
}

  name          = "alias/${local.name_prefix}-cmk"
  target_key_id = aws_kms_key.main.key_id

resource "aws_launch_template" "app" {
  name_prefix   = "${local.name_prefix}-lt-"
  image_id      = data.aws_ssm_parameter.al2023_ami.value
  instance_type = var.instance_type

  iam_instance_profile {
    name = aws_iam_instance_profile.ec2.name
  }

  vpc_security_group_ids = [aws_security_group.app.id]

  # No key pair: SSM Session Manager only

  # CKV_AWS_341: hop limit must be <= 1
  metadata_options {
    http_endpoint               = "enabled"
    http_tokens                 = "required" # IMDSv2
    http_put_response_hop_limit = 1
  }

  monitoring {
    enabled = true
  }

  block_device_mappings {
    device_name = "/dev/xvda"

    ebs {
      encrypted   = true
      kms_key_id  = aws_kms_key.main.arn
      volume_size = 20
      volume_type = "gp3"
    }
  }

  user_data = base64encode(<<-EOF
    #!/bin/bash
    set -euo pipefail

    # Simple health endpoint (for ALB later). Not exposed publicly.
    dnf -y install python3 || true
    nohup python3 -m http.server 8080 --bind 0.0.0.0 >/var/log/http.log 2>&1 &
  EOF
  )

  tag_specifications {
    resource_type = "instance"
    tags = {
      Name = "${local.name_prefix}-app"
    }
  }
}
